		modifier = {
		}

		should_start_with = {
			has_government = tribal_government
		}

		can_keep = { 
			trigger_if = {
				limit = {
					is_independent_ruler = yes
				}
				custom_tooltip = {
					text = mos_gov_any
					OR = {
						has_government = administrative_government
						has_government = feudal_government
						has_government = clan_government
						has_government = theocracy_government
						has_government = republican_government
						has_government = tribal_government
					}
				}
			}
			trigger_if = {
				limit = {
					is_independent_ruler = no
				}
				liege = { has_realm_law = mos_economic_reforms_major_1 }
			}
		}

		can_pass = {
			trigger_if = {
				limit = {
					is_independent_ruler = yes
				}
				custom_tooltip = {
					text = mos_reforms_cooldown_desc
					NOT = { has_variable = mos_reforms_cooldown }
				}
				has_realm_law = mos_economic_reforms_major_2
			}
			trigger_if = {
				limit = {
					is_independent_ruler = no
					liege = { has_realm_law = mos_economic_reforms_major_1 }
				}
				custom_tooltip = {
					text = imperial_bureaucracy_pending_law_change
					always = no
				}
			}
		}
		
		pass_cost = {
			gold = 0
		}
		
		on_pass = { 
			mos_major_econ_reform_change_cleanup_effect_1 = yes
			mos_pass_reform_effect = yes
		}
	}
